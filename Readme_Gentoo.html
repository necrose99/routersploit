Manual Install:
  1.     su -s       < for root term/xterm....>
  2.     cd /usr/src
  3.     git clone https://github.com/reverse-shell/routersploit.git
  4.     cd /usr/src/routersploit 
  5.     mkdir -p /opt/routersploit 
  6.     cp    /usr/src/routersploit/rsf.py   /opt/routersploit/rsf.py
  7.     cp -r /usr/src/routersploit/routersploit   /opt/routersploit
  8.     ln -s /opt/routersploit/rsf.py  /usr/bin/rsf/
